Perbedaan antara argumen fungsi dan parameter fungsi

    // nama disini adalah sebuah parameter fungsi
    function hallo( nama ) {
        return "hallo " + nama;
    }
    
    // keripix disini adalah sebuah argumen fungsi
    hallo("keripix");
